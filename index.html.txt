<script>
// Lightbox
function openLightbox(element) {
    const img = element.querySelector('img');
    document.getElementById('lightboxImg').src = img.src;
    document.getElementById('lightbox').classList.add('active');
}

function closeLightbox() {
    document.getElementById('lightbox').classList.remove('active');
}

// Donation Form
function submitDonation(e) {
    e.preventDefault();
    const name = document.getElementById('donorName').value;
    const amount = document.getElementById('donationAmount').options[document.getElementById('donationAmount').selectedIndex].text.split(' ')[0];
    const msg = document.getElementById('donationMessage');
    msg.innerHTML = '<i class="fas fa-check-circle"></i> Thank you ' + name + '! Your donation of LKR ' + amount + ' will help families in need.';
    msg.classList.add('show');
    document.getElementById('donationForm').reset();
    setTimeout(() => msg.classList.remove('show'), 5000);
}

// Volunteer Form
function submitVolunteer(e) {
    e.preventDefault();
    const name = document.getElementById('volName').value;
    const msg = document.getElementById('volunteerMessage');
    msg.innerHTML = '<i class="fas fa-check-circle"></i> Thank you ' + name + '! We will contact you soon about volunteer opportunities.';
    msg.classList.add('show');
    document.getElementById('volunteerForm').reset();
    setTimeout(() => msg.classList.remove('show'), 5000);
}
